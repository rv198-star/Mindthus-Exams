如果是常见 React 列表页，新增导出按钮大概会是这样：

```tsx
function ModuleToolbar({ filters }: { filters: ModuleFilters }) {
  const [exporting, setExporting] = useState(false);

  const handleExport = async () => {
    try {
      setExporting(true);

      const blob = await exportModuleData(filters);

      const url = window.URL.createObjectURL(blob);
      const link = document.createElement("a");
      link.href = url;
      link.download = "module-export.csv";
      link.click();

      window.URL.revokeObjectURL(url);
    } finally {
      setExporting(false);
    }
  };

  return (
    <Button onClick={handleExport} disabled={exporting}>
      {exporting ? "导出中..." : "导出"}
    </Button>
  );
}
```

接口封装：

```ts
export async function exportModuleData(filters: ModuleFilters): Promise<Blob> {
  const response = await api.get("/module/export", {
    params: filters,
    responseType: "blob",
  });

  return response.data;
}
```
